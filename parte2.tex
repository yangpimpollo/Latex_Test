     \item Sea $X$ una variable aleatorias cuya función de distribución $F$ es una función continua en linea recta.      \\
    \textbf{Solución:}
     $$Y=F(X)\sim U[0,1]~~~~~~Y=F(X)=P(X\le x)\in [0,1]    $$
     $como~ F(y)=(y-a)/(b-a)~para ~ y\in [a,b] $
     $$
P(Y\le 0)=P(F(x)\le 0)=P(P(X\le x)\le 0)=0$$$$
P(Y\le 1)=P(F(x)\le 1)=P(P(X\le x)\le 1)=1$$$$
P(Y\le y)=P(F(x)\le F(y))$$$$P(\int_{-\infty }^{x}f_x(t)dt-\int_{-\infty }^{y}f_x(t)dt)\le 0)=P(X\le y)=F(y)=y$$
     $$F_Y(y)=P(Y\le y)=\left\lbrace\begin{array}{c}
0 ~~~~~~ si~ y< 0~~~~~~\\ 
y ~~~~~~ si~ 0\le  y \le 1~~~~~~\\ 
1 ~~~~~~ si~ y>1~~~~~~\\ 
 \end{array}\right.$$

        \item Las variables $X,Y~y~Z$ son independientes e idénticamente distribuidos, y la función de distribución común $F$ es continua cual es $P(X<Y<Z)$  justifique su respuesta. (sugerencia:ejercicio anterior)
    \end{enumerate}
